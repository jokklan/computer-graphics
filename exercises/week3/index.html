Model View Matrix

        t = degreest × π
        p = degreesp × π

        at =  (0, 0, 0)
        up =  (0, 1, 0)
        eye = (r × sin(p), r = sin(t), r × cos(p))

        v = |at - eye|  # view direction vector
        n = |v x up|    # perpendicular vector
        u = |n x v|     # "new" up vector

        v = - v
      		
    

    

       MV =
      
        

          
            

              n
              - (n · eye)
              0
              1
            

            

              u
              - (u · eye)
              0
              1
            

            

              v
              - (v · eye)
              0
              1
            

            

              0
              0
              0
              1
            

          
        

      		
    

    

      
        
Perspective Matrix

        w = right - left
        h = top - bottom
        d = far - near
      		
    

    

      P =
      
        

          
            

              2 / w
              0
              0
              - (left +  right) / w
            

            

              0
              2 / h
              0
              - (top +  bottom) / h
            

            

              0
              0
              - 2 / d
              - (near +  far) / d
            

            

              0
              0
              0
              1
            

          
        

      		
    

    

      
        
One-point perspective cube

        degreest = 0
        degreesp = 0
        far = - near = 10
        top = - bottom = 3
        right = 3 + 2 = 5
        left = - 3 + 2 = 1

        CTM1 = P × MP
